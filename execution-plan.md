## 🎯 نظرة عامة على المشروع {#overview}

### الفكرة الأساسية

منصة SaaS متخصصة في حجز اليخوت للسوق السعودي، تتيح لكل مالك يخت إنشاء متجره الخاص وإدارته بالكامل، مع نظام حجوزات احترافي ودفع آمن.

### المشكلة التي نحلها

- ✅ **حراج ويلّه**: مجرد إعلانات بدون نظام حجز حقيقي
- ✅ **WhatsApp**: تواصل عشوائي وغير احترافي
- ✅ **طرق تقليدية**: صعوبة المقارنة والحجز الفوري
- ✅ **عدم الثقة**: لا يوجد نظام دفع آمن ومضمون

### الحل

منصة متكاملة توفر:

1. متجر مستقل لكل مالك يخت
2. نظام حجز وتقويم احترافي
3. دفع إلكتروني آمن
4. إدارة كاملة للحجوزات والعملاء
5. تجربة مستخدم راقية

## 💰 نموذج العمل {#business-model}

### مصادر الإيرادات

#### 1. العمولة على الحجوزات (الأساسي)

- **النسبة**: 10-15% من قيمة كل حجز
- **متى يتم الخصم**: عند إتمام الدفع
- **التحويل للمشغل**: خلال 7-14 يوم

#### 2. خطط اشتراك للمشغلين (اختياري - مستقبلاً)

- **Basic**: مجاني - عمولة 15%
- **Pro**: 299 ريال/شهر - عمولة 10% + ميزات إضافية
- **Enterprise**: 599 ريال/شهر - عمولة 8% + أولوية في البحث

#### 3. خدمات إضافية

- **إعلانات مميزة**: 500-2000 ريال/شهر
- **تصوير احترافي**: 1500 ريال/جلسة
- **إدارة التواصل الاجتماعي**: 1000 ريال/شهر

### مثال حسابي

```
حجز يخت:
- قيمة الحجز: 3000 ريال
- العمولة (12%): 360 ريال
- يحصل المشغل: 2640 ريال
- تحصل المنصة: 360 ريال

إذا تم 100 حجز شهرياً:
- إيرادات شهرية: 36,000 ريال
```

```typescript
// lib/auth.ts
import { NextAuthOptions } from "next-auth"
import CredentialsProvider from "next-auth/providers/credentials"
import { prisma } from "./db"

export const authOptions: NextAuthOptions = {
  providers: [
    CredentialsProvider({
      credentials: {
        phone: { label: "Phone", type: "text" },
        password: { label: "Password", type: "password" }
      },
      async authorize(credentials) {
        const user = await prisma.user.findUnique({
          where: { phone: credentials?.phone }
        })
        
        if (user && verifyPassword(credentials.password, user.password)) {
          return {
            id: user.id,
            name: user.name,
            email: user.email,
            role: user.role
          }
        }
        return null
      }
    })
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.role = user.role
      }
      return token
    },
    async session({ session, token }) {
      if (session?.user) {
        session.user.role = token.role
      }
      return session
    }
  }
}
```

```typescript
// lib/availability.ts
export async function checkAvailability(
  yachtId: string,
  date: Date,
  startTime: string,
  duration: number
): Promise<boolean> {
  // 1. Check if yacht is active
  const yacht = await prisma.yacht.findUnique({
    where: { id: yachtId },
    select: { status: true, isActive: true }
  })
  
  if (!yacht?.isActive || yacht.status !== 'AVAILABLE') {
    return false
  }

  // 2. Check blocked dates
  const blocked = await prisma.availability.findFirst({
    where: {
      yachtId,
      isBlocked: true,
      OR: [
        { date },
        { dayOfWeek: getDayOfWeek(date) }
      ]
    }
  })
  
  if (blocked) return false

  // 3. Check existing bookings
  const endTime = addHours(startTime, duration)
  
  const conflictingBooking = await prisma.booking.findFirst({
    where: {
      yachtId,
      date,
      status: { in: ['CONFIRMED', 'PAID', 'IN_PROGRESS'] },
      OR: [
        {
          AND: [
            { startTime: { lte: startTime } },
            { endTime: { gt: startTime } }
          ]
        },
        {
          AND: [
            { startTime: { lt: endTime } },
            { endTime: { gte: endTime } }
          ]
        },
        {
          AND: [
            { startTime: { gte: startTime } },
            { endTime: { lte: endTime } }
          ]
        }
      ]
    }
  })
  
  return !conflictingBooking
}
```

#### اليوم 6-7: Price Calculation

```typescript
// lib/pricing.ts
export async function calculateBookingPrice(
  yachtId: string,
  date: Date,
  duration: number,
  addons: { id: string, quantity: number }[]
): Promise<PriceBreakdown> {
  const yacht = await prisma.yacht.findUnique({
    where: { id: yachtId },
    include: {
      pricingRules: {
        where: { isActive: true },
        orderBy: { priority: 'desc' }
      },
      store: true
    }
  })

  let basePrice = yacht.basePrice * duration

  // Apply pricing rules
  for (const rule of yacht.pricingRules) {
    if (isRuleApplicable(rule, date)) {
      basePrice = applyRule(basePrice, rule)
    }
  }

  // Calculate addons
  const addonsData = await prisma.addon.findMany({
    where: { id: { in: addons.map(a => a.id) } }
  })
  
  const addonsCost = addonsData.reduce((sum, addon) => {
    const quantity = addons.find(a => a.id === addon.id)?.quantity || 1
    return sum + (addon.price * quantity)
  }, 0)

  const subtotal = basePrice + addonsCost
  const platformFee = subtotal * (yacht.store.commissionRate / 100)
  const tax = (subtotal + platformFee) * 0.15 // VAT 15%
  const total = subtotal + platformFee + tax

  return {
    basePrice,
    addonsCost,
    subtotal,
    platformFee,
    tax,
    total
  }
}
```

```typescript
// app/api/payment/create/route.ts
import { NextRequest, NextResponse } from 'next/server'
import Moyasar from '@moyasar/moyasar'

const moyasar = new Moyasar({
  apiKey: process.env.MOYASAR_API_KEY!
})

export async function POST(request: NextRequest) {
  const { bookingId } = await request.json()
  
  const booking = await prisma.booking.findUnique({
    where: { id: bookingId },
    include: { yacht: true, store: true }
  })

  try {
    const payment = await moyasar.payment.create({
      amount: booking.total * 100, // Convert to halalas
      currency: 'SAR',
      description: `حجز يخت ${booking.yacht.name}`,
      callback_url: `${process.env.NEXT_PUBLIC_URL}/payment/callback`,
      source: {
        type: 'creditcard'
      },
      metadata: {
        bookingId: booking.id,
        storeId: booking.storeId
      }
    })

    return NextResponse.json({ paymentUrl: payment.source.transaction_url })
  } catch (error) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }
}
```

1. Payment Callback Handler

```typescript
// app/api/payment/callback/route.ts
export async function GET(request: NextRequest) {
  const searchParams = request.nextUrl.searchParams
  const paymentId = searchParams.get('id')
  
  const payment = await moyasar.payment.fetch(paymentId)
  
  if (payment.status === 'paid') {
    await prisma.booking.update({
      where: { id: payment.metadata.bookingId },
      data: {
        status: 'PAID',
        paymentStatus: 'PAID'
      }
    })
    
    // Create payment record
    await prisma.payment.create({
      data: {
        transactionId: payment.id,
        bookingId: payment.metadata.bookingId,
        amount: payment.amount / 100,
        method: payment.source.type,
        provider: 'MOYASAR',
        status: 'PAID',
        paidAt: new Date()
      }
    })
    
    // Send confirmation email/SMS
    await sendBookingConfirmation(payment.metadata.bookingId)
  }
  
  return redirect(`/booking/${payment.metadata.bookingId}/success`)
}
```

```typescript
// middleware.ts
import { withAuth } from "next-auth/middleware"

export default withAuth({
  callbacks: {
    authorized: ({ req, token }) => {
      const path = req.nextUrl.pathname
      
      // Public routes
      if (path.startsWith('/yachts') || path === '/') {
        return true
      }
      
      // Protected routes
      if (path.startsWith('/dashboard/admin')) {
        return token?.role === 'ADMIN'
      }
      
      if (path.startsWith('/dashboard/operator')) {
        return token?.role === 'OPERATOR' || token?.role === 'ADMIN'
      }
      
      if (path.startsWith('/account') || path.startsWith('/bookings')) {
        return !!token
      }
      
      return false
    },
  },
})

export const config = {
  matcher: ['/dashboard/:path*', '/account/:path*', '/bookings/:path*']
}
```
